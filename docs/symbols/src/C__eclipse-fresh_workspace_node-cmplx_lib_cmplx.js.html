  9 /**
 10  * Abstract base class for complex number. This class is extended by
 11  * <code>Arith</code> and <code>Polar</code>. It's not intended to
 12  * be used directly.  
 13  * @name Cmplx
 14  * @constructor
 15  */
 16 function Cmplx() {
 17 	if (this instanceof Cmplx) {
 18 	} else return new Cmplx();
 19 }

 65 /**
 66  * A complex number in arithmetic (a + i b) form.
 67  * @name Arith
 68  * @constructor
 69  * @extends Cmplx
 70  * @param {Number} real
 71  * @param {Number} imaginary
 72  */
 73 function Arith(real, imaginary) {
 74 	if (this instanceof Arith) {
 75 		this.real = real;
 76 		this.imaginary = imaginary;
 77 		this.__defineGetter__('radius', function () {
 78 			return Math.sqrt(this.real*this.real+this.imaginary*this.imaginary);
 79 		});
 80 		this.__defineGetter__('angle', function () {
 81 			return Math.atan2(this.imaginary, this.real);
 82 		});
 83 	} else return new Arith(real, imaginary);
 84 }

155 /**
156  * @name Arith.add
157  * @function
158  * @param {Cmplx} lhs
159  * @param {Cmplx} rhs
160  * @returns {Arith}
161  */
162 Arith.add = function (lhs, rhs) {
163 	return Arith(lhs.real+rhs.real, lhs.imaginary+rhs.imaginary);
164 };
165 
166 /**
167  * @name Arith.sub
168  * @function
169  * @param {Cmplx} lhs
170  * @param {Cmplx} rhs
171  * @returns {Arith}
172  */
173 Arith.sub = function (lhs, rhs) {
174 	return Arith(lhs.real-rhs.real, lhs.imaginary-rhs.imaginary);
175 };
176 
177 /**
178  * @name Arith.mul
179  * @function
180  * @param {Cmplx} lhs
181  * @param {Cmplx} rhs
182  * @returns {Arith}
183  */
184 Arith.mul = function (lhs, rhs) {
185 	return Arith(lhs.real*rhs.real-lhs.imaginary*rhs.imaginary, lhs.real*rhs.imaginary+rhs.real*lhs.imaginary);
186 };
187 
188 /**
189  * @name Arith.div
190  * @function
191  * @param {Cmplx} lhs
192  * @param {Cmplx} rhs
193  * @returns {Arith}
194  */
195 Arith.div = function (lhs, rhs) {
196 	var d = rhs.real*rhs.real+rhs.imaginary*rhs.imaginary;
197 	return Arith((lhs.real*rhs.real+lhs.imaginary*rhs.imaginary)/d, (-lhs.real*rhs.imaginary+rhs.real*lhs.imaginary)/d);
198 };

220 /**
221  * A complex number in polar (r e ^ (i p)) form.
222  * @name Polar
223  * @constructor
224  * @extends Cmplx
225  * @param {Number} radius
226  * @param {Number} angle
227  */
228 function Polar(radius, angle) {
229 	if (this instanceof Polar) {
230 		this.radius = radius;
231 		this.angle = angle;
232 		this.__defineGetter__('real', function () {
233 			return this.radius*Math.cos(this.angle);
234 		});
235 		this.__defineGetter__('imaginary', function () {
236 			return this.radius*Math.sin(this.angle);
237 		});
238 	} else return new Polar(radius, angle);
239 }
